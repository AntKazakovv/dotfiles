```
export const $modules = {
    tournaments: {
        components: {
            'wlc-tournament-detail': {
                prizesParams: {
                    showMore: {
                        use: false, // set true to trim prizeboard
                        rowLimit: 10, // number of displayed rows
                    },
                },
            },
        },
    },
};
```
