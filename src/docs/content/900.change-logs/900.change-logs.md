Также введен новый параметр useFor, для выборочного включения категорий в Депозите или Выводе (по умолчанию: 'deposit').
> XX.finances.config.ts
```
export const $finances: IFinancesConfig = {
    paySystemCategories: {
        use: true | false,
        useFor: 'withdraw' | 'deposit' | 'both',
        desktopMenuType: 'select' | 'menu' | 'blocks',
    },
};
```
