# 3.2.1. Базовая конфигурация

Основная конфигурация проекта WLC. (см. [интерфейс](../../../interfaces/IBaseConfig.html) объекта).

## Пример конфигурирования модальных окон

* [Интерфейс](../../../interfaces/IModalList.html)
* Описание модальных окно находится по этому [пути](https://vcsxa.egamings.com/wlc/wlc-engine/-/blob/master/src/modules/core/components/modal/modal.params.ts) в переменной __MODALS_LIST__

```typescript
export const $base: IBaseConfig = {
    site: {
        restrictRegistration: false, // disable registation
    },
    modals: {
        login: {
            config: {
                id: 'login',
                modifier: 'login-sign-up',
                modalTitle: gettext('Sign in now!'),
                component: TabSwitcherComponent,
                size: 'as',
                showFooter: false,
                dismissAll: true,
                componentParams: {
                    themeMod: 'with-banner',
                    tabs: {
                        'signIn': {
                            name: gettext('Sign in'),
                            startTab: 'signIn',
                            component: 'user.wlc-sign-in-form',
                            modifier: 'login',
                            modalId: 'login',
                        },
                        'signUp': {
                            name: gettext('Sign up'),
                            component: 'core.wlc-steps',
                            theme: 'signInUp',
                            modifier: 'signup',
                            modalId: 'signup',
                        },
                    },
                },
            },
        },
    },
};
```
