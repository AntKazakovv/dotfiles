# 3.2.6.1 Управление через конфигурацию в проекте

## Где задаются настройки

Создаем на уровне проекта файл `/config/frontend/menu.config.json`.

В нем создаем следующее содержимое

```json
{
    "menuSettings": {
        ... тут будут настройки меню
    },
    "categories": {
        ... тут будут настройки категорий
    },
}
```

## Настройка меню

## Настройка категорий

### Из чего состоят настройки

```json
"categories": {
    // slug категории, для которой будут заданы настройки
    "casino": {
        // вариант представления игр в каталоге в виде игровых блоков
        "view": "blocks"
    },
    "blackjacks": {
        // вариант представления игр в каталоге в виде общего списка игр
        "view": "all"
    },
    "livecasino": {
        // вариант представления игр в каталоге в виде игровых блоков с доп возмоджностями настройки
        "view": "restricted-blocks",
        "blocks": {
            "popular": {
                "order": 7 // в каком порядке вывести игровой блок
            },
            "new": {
                "order": 6
            },
            "html5": {
                "order": 5,
                "gameRows": { // настройки строк в сетке игр
                    "desktop": 1, // сколько строк отображать в desktop версии
                    "mobile": 3 // сколько строк отображать в mobile версии
                },
                "showType": "slide-arrows" // отобразить сетку игр в виде свайпера. Доступны варианты 'btn-load-more' | 'slide-arrows'
            },
            "blackjacks": {
                "order": 4,
                "gameRows": {
                    "desktop": 1,
                    "mobile": 3
                },
                "showType": "btn-load-more" // отобразить сетку игр с кнопкой "Показать еще"
            },
            "videopoker": {
                "order": 3
            },
            "jackpots": {
                "order": 1
            },
            "slots": {
                "order": 2
            }
        }
    },
}
```
